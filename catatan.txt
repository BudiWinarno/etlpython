CREATE TABLE item_agent_mappings (

    id SERIAL PRIMARY KEY,

    agent_id INTEGER NOT NULL,

    kode_sku_agent VARCHAR(100) NOT NULL,

    kode_sku_jim VARCHAR(100) NOT NULL,

    nama_sku_jim VARCHAR(255) NOT NULL,

    item_box NUMERIC(10,2) NOT NULL,

    item_group VARCHAR(100),

    is_active BOOLEAN NOT NULL DEFAULT TRUE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT uq_item_agent_mapping
        UNIQUE (agent_id, kode_sku_agent)

)
